function handleScrollAnimation() {
    const elements = document.querySelectorAll('.fade-in-element');
    
    elements.forEach(element => {
        const elementTop = element.getBoundingClientRect().top;
        const elementVisible = 150; // Trigger when element is 150px from viewport
        
        if (elementTop < window.innerHeight - elementVisible) {
            element.classList.add('fade-in');
        }
    });
}

// About section card slider functionality
const aboutCardsRow = document.querySelector('.about-cards-row');
if (aboutCardsRow) {
    const cards = aboutCardsRow.querySelectorAll('.col-md-6.fade-in-element');
    const leftBtn = document.getElementById('about-arrow-left');
    const rightBtn = document.getElementById('about-arrow-right');
    
    let currentPage = 0;
    const cardsPerPage = 2;
    const totalPages = Math.ceil(cards.length / cardsPerPage);
    
    function showCards(page, direction) {
        // Hide all cards
        cards.forEach(card => {
            card.classList.add('about-card-hidden');
            card.classList.remove('slide-in-right', 'slide-in-left', 'slide-out-left', 'slide-out-right');
        });
        
        // Show new cards with slide-in animation
        for (let i = page * cardsPerPage; i < (page + 1) * cardsPerPage && i < cards.length; i++) {
            cards[i].classList.remove('about-card-hidden');
            if (direction === 'right') {
                cards[i].classList.add('slide-in-right');
            } else if (direction === 'left') {
                cards[i].classList.add('slide-in-left');
            }
        }
    }
    
    function updateCards() {
        // Show only current page cards, no animation
        cards.forEach((card, index) => {
            card.classList.remove('slide-in-right', 'slide-in-left', 'slide-out-left', 'slide-out-right');
            if (index >= currentPage * cardsPerPage && index < (currentPage + 1) * cardsPerPage) {
                card.classList.remove('about-card-hidden');
            } else {
                card.classList.add('about-card-hidden');
            }
        });
        
        if (leftBtn) leftBtn.disabled = currentPage === 0;
        if (rightBtn) rightBtn.disabled = currentPage === totalPages - 1;
    }
    
    if (leftBtn) {
        leftBtn.addEventListener('click', function() {
            if (currentPage > 0) {
                // Animate current cards out to right
                const currentCards = Array.from(cards).filter((card, index) => 
                    index >= currentPage * cardsPerPage && index < (currentPage + 1) * cardsPerPage
                );
                currentCards.forEach(card => {
                    card.classList.remove('slide-in-right', 'slide-in-left', 'slide-out-left', 'slide-out-right');
                    card.classList.add('slide-out-right');
                });
                setTimeout(() => {
                    currentPage--;
                    showCards(currentPage, 'left');
                    setTimeout(() => {
                        updateCards();
                    }, 450);
                }, 450);
            }
        });
    }
    
    if (rightBtn) {
        rightBtn.addEventListener('click', function() {
            if (currentPage < totalPages - 1) {
                // Animate current cards out to left
                const currentCards = Array.from(cards).filter((card, index) => 
                    index >= currentPage * cardsPerPage && index < (currentPage + 1) * cardsPerPage
                );
                currentCards.forEach(card => {
                    card.classList.remove('slide-in-right', 'slide-in-left', 'slide-out-left', 'slide-out-right');
                    card.classList.add('slide-out-left');
                });
                setTimeout(() => {
                    currentPage++;
                    showCards(currentPage, 'right');
                    setTimeout(() => {
                        updateCards();
                    }, 450);
                }, 450);
            }
        });
    }
    
    // Initialize
    updateCards();
}

// Real-time Date and Time Display
function updateRealTime() {
    const now = new Date();
    
    // Format date as "20 June 2025"
    const day = now.getDate();
    const monthNames = [
        "January", "February", "March", "April", "May", "June",
        "July", "August", "September", "October", "November", "December"
    ];
    const month = monthNames[now.getMonth()];
    const year = now.getFullYear();
    
    // Format time as "8:00 AM"
    let hours = now.getHours();
    const minutes = now.getMinutes().toString().padStart(2, '0');
    const ampm = hours >= 12 ? 'PM' : 'AM';
    hours = hours % 12;
    hours = hours ? hours : 12; // the hour '0' should be '12'
    
    // Format day of week as "Monday"
    const dayNames = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"];
    const dayOfWeek = dayNames[now.getDay()];
    
    // Combine all parts: "20 June 2025, 8:00 AM Monday"
    const formattedDateTime = `${day} ${month} ${year}, ${hours}:${minutes} ${ampm} ${dayOfWeek}`;
    
    // Update the display
    const realTimeElement = document.getElementById('realTimeDisplay');
    if (realTimeElement) {
        realTimeElement.textContent = formattedDateTime;
    }
}

// Update immediately when page loads
updateRealTime();

// Update every second (1000 milliseconds)
setInterval(updateRealTime, 1000);

        // Scroll-triggered fade-in animation
        function handleScrollAnimation() {
            const elements = document.querySelectorAll('.fade-in-element');
            
            elements.forEach(element => {
                const elementTop = element.getBoundingClientRect().top;
                const elementVisible = 150; // Trigger when element is 150px from viewport
                
                if (elementTop < window.innerHeight - elementVisible) {
                    element.classList.add('fade-in');
                }
            });
        }

        // Add event listeners
        window.addEventListener('scroll', handleScrollAnimation);
        window.addEventListener('load', handleScrollAnimation);

        // Optional: Re-trigger animation on window resize

        // About section card slider
        const aboutCardsRow = document.querySelector('.about-cards-row');
        const cards = aboutCardsRow.querySelectorAll('.col-md-6.fade-in-element');
        const leftBtn = document.getElementById('about-arrow-left');
        const rightBtn = document.getElementById('about-arrow-right');
        
        let currentPage = 0;
        const cardsPerPage = 2;
        const totalPages = Math.ceil(cards.length / cardsPerPage);
        
        function showCards(page, direction) {
            // Hide all cards
            cards.forEach(card => {
                card.classList.add('about-card-hidden');
                card.classList.remove('slide-in-right', 'slide-in-left', 'slide-out-left', 'slide-out-right');
            });
            // Show new cards with slide-in animation
            for (let i = page * cardsPerPage; i < (page + 1) * cardsPerPage && i < cards.length; i++) {
                cards[i].classList.remove('about-card-hidden');
                if (direction === 'right') {
                    cards[i].classList.add('slide-in-right');
                } else if (direction === 'left') {
                    cards[i].classList.add('slide-in-left');
                }
            }
        }
        
        function updateCards() {
            // Show only current page cards, no animation
            cards.forEach((card, index) => {
                card.classList.remove('slide-in-right', 'slide-in-left', 'slide-out-left', 'slide-out-right');
                if (index >= currentPage * cardsPerPage && index < (currentPage + 1) * cardsPerPage) {
                    card.classList.remove('about-card-hidden');
                } else {
                    card.classList.add('about-card-hidden');
                }
            });
            leftBtn.disabled = currentPage === 0;
            rightBtn.disabled = currentPage === totalPages - 1;
        }
        
        leftBtn.addEventListener('click', function() {
            if (currentPage > 0) {
                // Animate current cards out to right
                const currentCards = Array.from(cards).filter((card, index) => 
                    index >= currentPage * cardsPerPage && index < (currentPage + 1) * cardsPerPage
                );
                currentCards.forEach(card => {
                    card.classList.remove('slide-in-right', 'slide-in-left', 'slide-out-left', 'slide-out-right');
                    card.classList.add('slide-out-right');
                });
                setTimeout(() => {
                    currentPage--;
                    showCards(currentPage, 'left');
                    setTimeout(() => {
                        updateCards();
                    }, 450);
                }, 450);
            }
        });
        
        rightBtn.addEventListener('click', function() {
            if (currentPage < totalPages - 1) {
                // Animate current cards out to left
                const currentCards = Array.from(cards).filter((card, index) => 
                    index >= currentPage * cardsPerPage && index < (currentPage + 1) * cardsPerPage
                );
                currentCards.forEach(card => {
                    card.classList.remove('slide-in-right', 'slide-in-left', 'slide-out-left', 'slide-out-right');
                    card.classList.add('slide-out-left');
                });
                setTimeout(() => {
                    currentPage++;
                    showCards(currentPage, 'right');
                    setTimeout(() => {
                        updateCards();
                    }, 450);
                }, 450);
            }
        });
        
        // Initialize
        updateCards();
        
    </script>

    <script>
        function updateRealTime() {
            const now = new Date();
            
            // Format date as "20 June 2025"
            const day = now.getDate();
            const monthNames = [
                "January", "February", "March", "April", "May", "June",
                "July", "August", "September", "October", "November", "December"
            ];
            const month = monthNames[now.getMonth()];
            const year = now.getFullYear();
            
            // Format time as "8:00 AM"
            let hours = now.getHours();
            const minutes = now.getMinutes().toString().padStart(2, '0');
            const ampm = hours >= 12 ? 'PM' : 'AM';
            hours = hours % 12;
            hours = hours ? hours : 12; // the hour '0' should be '12'
            
            // Format day of week as "Monday"
            const dayNames = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"];
            const dayOfWeek = dayNames[now.getDay()];
            
            // Combine all parts: "20 June 2025, 8:00 AM Monday"
            const formattedDateTime = `${day} ${month} ${year}, ${hours}:${minutes} ${ampm} ${dayOfWeek}`;
            
            // Update the display
            const realTimeElement = document.getElementById('realTimeDisplay');
            if (realTimeElement) {
                realTimeElement.textContent = formattedDateTime;
            }
        }
        
        // Update immediately when page loads
        updateRealTime();
        
        // Update every second (1000 milliseconds)
        setInterval(updateRealTime, 1000);
    </script>
